### 7.2.2 Voices 页面完整示例

**服务端（page.tsx）：**

```typescript
// src/app/(dashboard)/voices/page.tsx
import { prefetch, trpc, HydrateClient } from "@/trpc/server";
import { voicesSearchParamsCache } from "@/features/voices/lib/params";

export default async function VoicesPage({
  searchParams,
}: {
  searchParams: Promise<SearchParams>;
}) {
  const { query } = await voicesSearchParamsCache.parse(searchParams);

  // 1. 预取数据
  prefetch(trpc.voices.getAll.queryOptions({ query }));

  // 2. 脱水传给客户端
  return (
    <HydrateClient>
      <VoicesView />
    </HydrateClient>
  );
}
```

**客户端（voices-view.tsx）：**

```typescript
// src/features/voices/views/voices-view.tsx
export function VoicesView() {
  const trpc = useTRPC();
  const [query] = useQueryState("query");

  // 3. 使用 useSuspenseQuery（数据已预取）
  const { data, isLoading } = useSuspenseQuery(
    trpc.voices.getAll.queryOptions({ query })
  );

  if (isLoading) return <Skeleton />;

  return (
    <div>
      {data.custom.map((voice) => (
        <VoiceCard key={voice.id} voice={voice} />
      ))}
    </div>
  );
}
```

## 7.4 性能优化技巧

### 7.4.1 staleTime 优化

```typescript
// 默认配置
staleTime: 30 * 1000,  // 30秒内不重新请求

// 针对不同数据设置不同 staleTime
const { data } = useQuery({
  queryKey: ['users'],
  queryFn: fetchUsers,
  staleTime: 5 * 60 * 1000,  // 静态数据设置5分钟
});
```

```typescript
// 组件中使用缓存
```

### 7.5.2 Mutation 最佳实践

```typescript
// ✅ 1. 成功后刷新相关数据
const mutation = useMutation(
  trpc.users.create.mutationOptions({
    onSuccess: () => {
      queryClient.invalidateQueries({
```

## 7.6 常见问题解决

### 7.6.1 数据不刷新

**问题**：修改数据后列表不更新

**解决**：

```typescript
// 确保调用 invalidateQueries
onSuccess: () => {
  queryClient.invalidateQueries({
    queryKey: trpc.voices.getAll.queryKey(),
  });
}
```

### 7.6.2 请求不发送

**问题**：useQuery 没有发送请求

**解决**：

```typescript
// 检查 enabled
const { data } = useQuery({
  queryKey: ['users'],
  queryFn: fetchUsers,
  enabled: true,  // 必须是 true
});
```

### 7.6.3 缓存冲突

**问题**：不同参数的数据互相覆盖

**解决**：

```typescript
// 确保 queryKey 包含所有参数
queryKey: ['users', { status, page, limit }]
```

### 7.6.4 SSR 数据丢失

**问题**：刷新页面后数据丢失

**解决**：

```typescript
// 确保 HydrationBoundary 正确配置
<HydrateClient>
  <Page />
</HydrateClient>
```

## 7.7 项目配置汇总

### 7.7.1 query-client.ts

```typescript
export function makeQueryClient() {
  return new QueryClient({
    defaultOptions: {
      queries: {
        staleTime: 30 * 1000,
      },
      dehydrate: {
        serializeData: superjson.serialize,
        shouldDehydrateQuery: (query) =>
          defaultShouldDehydrateQuery(query) ||
          query.state.status === 'pending',
      },
      hydrate: {
        deserializeData: superjson.deserialize,
      },
    },
  });
}
```

### 7.7.2 server.tsx

```typescript
export const getQueryClient = cache(makeQueryClient);

export function HydrateClient({ children }) {
  const queryClient = getQueryClient();
  return (
    <HydrationBoundary state={dehydrate(queryClient)}>
      {children}
    </HydrationBoundary>
  );
}

export function prefetch(queryOptions) {
  const queryClient = getQueryClient();
  void queryClient.prefetchQuery(queryOptions);
}
```
